# the below designates the "Date" column as the index BUT w/o the parse_dates command would come in as a string
# the parse_dates parameter will cause it to make best efforts to convert the string to a datetime
# note that there is more control allowed to the longer method but with well formatted dates this shorter way works
df2 = pd.read_csv('time_data/walmart_stock.csv', index_col='Date', parse_dates=True)
